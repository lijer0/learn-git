--git branch （创建分支）（--git branch -d xxx 删除分支）
--git switch （切换分支）（--git switch -c xxx /--git checkout -b xxx 创建并切换）
--git merge （合并分支;-no-ff -m 'xxx'）
--git tag （标签）
--git stash （暂存工作区）
--git pull 
--git rebase

KVM:
~sudo apt install qemu qemu-kvm libvirt-bin bridge-utils virt-manager
~sudo service libvirtd start
~sudo systemctl enable libvirtd.service
~sudo service libvirtd status
--A,桥接Bridge模式
--编辑/etc/network/interfaces，增加如下内容
	auto br0
	iface br0 inet dhcp  # 网桥使用DHCP模式，从DHCP服务器获取IP
	bridge_ports wlp3s0  # 网桥创建前连接外部的网卡名称，ifconfig命令查看有IP地址的就是
	bridge_stp on  # 避免数据链路出现死循环
	bridge_fd 0  # 将转发延迟设置为0
--sudo systemctl restart networking.service  #重启networking服务
*--编辑/etc/default/ufw允许ip转发
--DEFAULT_FORWARD_POLICY="ACCEPT"
--sudo systemctl restart ufw.service  #重启ufw服务
